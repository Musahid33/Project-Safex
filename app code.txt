        async function executePublicLiveSearch() {
            const selectedPlant = document.getElementById('publicSearchPlant').value;
            const searchId = document.getElementById('publicSearchEmpId').value.trim().toUpperCase();
            const resultBox = document.getElementById('publicResultBox');
            if(!searchId) return;
            resultBox.classList.remove('hidden');
            try {
                const { data } = await _supabase.from('employee_profiles').select('*').eq('employee_id', searchId);
                if(data && data[0] && data[0].active_plant_id === selectedPlant) {
                    resultBox.className = "rounded-xl p-3 bg-emerald-50 text-slate-700 text-left space-y-1.5";
                    resultBox.innerHTML = `✅ <strong>${data[0].full_name}</strong><br>Blood: ${data[0].blood_group}<br>Status: ${data[0].medical_fitness_status}`;
                } else {
                    resultBox.className = "rounded-xl p-3 bg-rose-100 text-rose-700";
                    resultBox.innerHTML = `🛑 Verification Boundary Error.`;
                }
            } catch(e) { resultBox.innerHTML = e.message; }
        }
